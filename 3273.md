### kotlin

> 투 포인터 사용

```kotlin
import java.util.*

fun main() = with(Scanner(System.`in`)) {
    val n = nextInt()
    val sequence = IntArray(n)
    for (i in 0 until n) sequence[i] = nextInt()
    val x = nextInt()

    sequence.sort()

    var start = 0
    var end = n - 1
    var result = 0

    while (start < end) {
        var sum = sequence[start] + sequence[end]

        if(sum == x) result += 1
        if(sum <= x) start += 1
        else end -= 1
    }

    print(result)
}
```

> 1500 ms 사용하여 통과 → 보완 필요

```kotlin
import java.util.*

fun main() = with(Scanner(System.`in`)) {
    val n = nextInt()
    val sequence = IntArray(n)

    for (i in 0 until n) sequence[i] = nextInt()

    val x = nextInt()
    var result = 0

    for (i in sequence.indices) {
        if (x - sequence[i] in 0..1000000 && sequence.any { it == x - sequence[i] }) result += 1
    }

    print(result / 2)
}
```
